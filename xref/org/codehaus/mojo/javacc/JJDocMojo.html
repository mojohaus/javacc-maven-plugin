41  /**
42   * <a href="https://javacc.dev.java.net/doc/JJDoc.html">JJDoc</a> takes a JavaCC parser specification and produces
43   * documentation for the BNF grammar. This mojo will search the source directory for all <code>*.jj</code> files and
44   * run JJDoc once for each file it finds. Each of these output files, along with an <code>index.html</code> file will
45   * be placed in the site directory (<code>target/site/jjdoc</code>), and a link will be created in the "Project
46   * Reports" menu of the generated site.
47   *
48   * @since 2.3
49   * @author <a href="mailto:pgier@redhat.com">Paul Gier</a>
50   * @see <a href="https://javacc.dev.java.net/doc/JJDoc.html">JJDoc Documentation</a>
51   */
52  @Mojo(name = "jjdoc")
53  @Execute(phase = LifecyclePhase.GENERATE_SOURCES)
54  public class JJDocMojo
55      extends AbstractMavenReport
56  {

215     /**
216      * Get the source directories that should be scanned for grammar files.
217      * 
218      * @return The source directories that should be scanned for grammar files, never <code>null</code>.
219      */
220     private File[] getSourceDirectories()
221     {
222         Set directories = new LinkedHashSet();
223         if ( this.sourceDirectories != null && this.sourceDirectories.length > 0 )
224         {
225             directories.addAll( Arrays.asList( this.sourceDirectories ) );
226         }
227         else
228         {
229             if ( this.defaultGrammarDirectoryJavaCC != null )
230             {
231                 directories.add( this.defaultGrammarDirectoryJavaCC );
232             }
233             if ( this.defaultGrammarDirectoryJJTree != null )
234             {
235                 directories.add( this.defaultGrammarDirectoryJJTree );
236             }
237             if ( this.defaultGrammarDirectoryJTB != null )
238             {
239                 directories.add( this.defaultGrammarDirectoryJTB );
240             }
241         }
242         return (File[]) directories.toArray( new File[directories.size()] );
243     }

283     {
284         File sourceDirs[] = getSourceDirectories();
285         for ( int i = 0; i < sourceDirs.length; i++ )
286         {
287             File sourceDir = sourceDirs[i];
288             String[] files = sourceDir.list();
289             if ( files != null && files.length > 0 )
290             {
291                 return true;
292             }
293         }
294         return false;

327     /**
328      * Create the header and title for the HTML report page.
329      * 
330      * @param bundle The resource bundle with the text.
331      * @param sink The sink for writing to the main report file.
332      */
333     private void createReportHeader( ResourceBundle bundle, Sink sink )
334     {
335         sink.head();
336         sink.title();
337         sink.text( bundle.getString( "report.jjdoc.title" ) );
338         sink.title_();
339         sink.head_();
340 
341         sink.body();
342 
343         sink.section1();
344         sink.sectionTitle1();
345         sink.text( bundle.getString( "report.jjdoc.title" ) );
346         sink.sectionTitle1_();
347         sink.text( bundle.getString( "report.jjdoc.description" ) );
348         sink.section1_();
349 
350         sink.lineBreak();
351         sink.table();
352         sink.tableRow();
353         sink.tableHeaderCell();
354         sink.text( bundle.getString( "report.jjdoc.table.heading" ) );
355         sink.tableHeaderCell_();
356         sink.tableRow_();
357     }

359     /**
360      * Create a table row containing a link to the JJDoc report for a grammar file.
361      * 
362      * @param sink The sink to write the report
363      * @param sourceDirectory The source directory of the grammar file.
364      * @param grammarFile The JavaCC grammar file.
365      * @param linkPath The path to the JJDoc output.
366      */
367     private void createReportLink( Sink sink, File sourceDirectory, File grammarFile, String linkPath )
368     {
369         sink.tableRow();
370         sink.tableCell();
371         if ( linkPath.startsWith( "/" ) )
372         {
373             linkPath = linkPath.substring( 1 );
374         }
375         sink.link( linkPath );
376         String grammarFileRelativePath = sourceDirectory.toURI().relativize( grammarFile.toURI() ).toString();
377         if ( grammarFileRelativePath.startsWith( "/" ) )
378         {
379             grammarFileRelativePath = grammarFileRelativePath.substring( 1 );
380         }
381         sink.text( grammarFileRelativePath );
382         sink.link_();
383         sink.tableCell_();
384         sink.tableRow_();
385     }

398     /**
399      * Creates a new facade to invoke JJDoc. Most options for the invocation are derived from the current values of the
400      * corresponding mojo parameters. The caller is responsible to set the input file and output file on the returned
401      * facade.
402      * 
403      * @return The facade for the tool invocation, never <code>null</code>.
404      */
405     private JJDoc newJJDoc()
406     {
407         JJDoc jjdoc = new JJDoc();
408         jjdoc.setLog( getLog() );
409         jjdoc.setGrammarEncoding( this.grammarEncoding );
410         jjdoc.setCssHref( this.cssHref );
411         jjdoc.setText( this.text );
412         jjdoc.setBnf( this.bnf );
413         jjdoc.setOneTable( Boolean.valueOf( this.oneTable ) );
414         return jjdoc;
415     }

417     /**
418      * Searches the specified source directory to find grammar files that can be documented.
419      * 
420      * @param sourceDirectory The source directory to scan for grammar files.
421      * @return An array of grammar infos describing the found grammar files or <code>null</code> if the source
422      *         directory does not exist.
423      * @throws MavenReportException If there is a problem while scanning for .jj files.
424      */
425     private GrammarInfo[] scanForGrammars( File sourceDirectory )
426         throws MavenReportException
427     {
428         if ( !sourceDirectory.isDirectory() )
429         {
430             return null;
431         }
432 
433         GrammarInfo[] grammarInfos;
434 
435         getLog().debug( "Scanning for grammars: " + sourceDirectory );
436         try
437         {
438             String[] includes = { "**/*.jj", "**/*.JJ", "**/*.jjt", "**/*.JJT", "**/*.jtb", "**/*.JTB" };
439             GrammarDirectoryScanner scanner = new GrammarDirectoryScanner();
440             scanner.setSourceDirectory( sourceDirectory );
441             scanner.setIncludes( includes );
442             scanner.scan();
443             grammarInfos = scanner.getIncludedGrammars();
444         }
445         catch ( Exception e )
446         {
447             throw new MavenReportException( "Failed to scan for grammars: " + sourceDirectory, e );
448         }
449         getLog().debug( "Found grammars: " + Arrays.asList( grammarInfos ) );
450 
451         return grammarInfos;
452     }

465     /**
466      * Compares grammar infos using their relative grammar file paths as the sort key.
467      */
468     private static class GrammarInfoComparator
469         implements Comparator
470     {
471 
472         /**
473          * The singleton instance of this comparator.
474          */
475         private static final GrammarInfoComparator INSTANCE = new GrammarInfoComparator();
476 
477         /**
478          * Gets the singleton instance of this class.
479          * 
480          * @return The singleton instance of this class.
481          */
482         public static GrammarInfoComparator getInstance()
483         {
484             return INSTANCE;
485         }
486 
487         /**
488          * Compares the path of two grammar files lexicographically.
489          * 
490          * @param o1 The first grammar info.
491          * @param o2 The second grammar info.
492          * @return A negative integer if the first grammar is considered "smaller", a positive integer if it is
493          *         considered "greater" and zero otherwise.
494          */
495         public int compare( Object o1, Object o2 )
496         {
497             int rel;
498 
499             GrammarInfo info1 = (GrammarInfo) o1;
500             String[] paths1 = info1.getRelativeGrammarFile().split( "\\" + File.separatorChar );
501 
502             GrammarInfo info2 = (GrammarInfo) o2;
503             String[] paths2 = info2.getRelativeGrammarFile().split( "\\" + File.separatorChar );
504 
505             int dirs = Math.min( paths1.length, paths2.length ) - 1;
506             for ( int i = 0; i < dirs; i++ )
507             {
508                 rel = paths1[i].compareToIgnoreCase( paths2[i] );
509                 if ( rel != 0 )
510                 {
511                     return rel;
512                 }
513             }
514 
515             rel = paths1.length - paths2.length;
516             if ( rel != 0 )
517             {
518                 return rel;
519             }
520 
521             return paths1[paths1.length - 1].compareToIgnoreCase( paths2[paths1.length - 1] );
522         }
523 
524     }

526     public void generate(Sink sink, Locale locale) throws MavenReportException {
527         createReportHeader( getBundle( locale ), sink );
528 
529         File[] sourceDirs = getSourceDirectories();
530         for ( int j = 0; j < sourceDirs.length; j++ )
531         {
532             File sourceDir = sourceDirs[j];
533             GrammarInfo[] grammarInfos = scanForGrammars( sourceDir );
534 
535             if ( grammarInfos == null )
536             {
537                 getLog().debug( "Skipping non-existing source directory: " + sourceDir );
538             }
539             else
540             {
541                 Arrays.sort( grammarInfos, GrammarInfoComparator.getInstance() );
542                 for ( int i = 0; i < grammarInfos.length; i++ )
543                 {
544                     GrammarInfo grammarInfo = grammarInfos[i];
545                     File grammarFile = grammarInfo.getGrammarFile();
546 
547                     String relativeOutputFileName = grammarInfo.getRelativeGrammarFile();
548                     relativeOutputFileName =
549                             relativeOutputFileName.replaceAll( "(?i)\\.(jj|jjt|jtb)$", getOutputFileExtension() );
550 
551                     File jjdocOutputFile = new File( getJJDocOutputDirectory(), relativeOutputFileName );
552 
553                     JJDoc jjdoc = newJJDoc();
554                     jjdoc.setInputFile( grammarFile );
555                     jjdoc.setOutputFile( jjdocOutputFile );
556                     try
557                     {
558                         jjdoc.run();
559                     }
560                     catch ( Exception e )
561                     {
562                         throw new MavenReportException( "Failed to create BNF documentation: " + grammarFile, e );
563                     }
564 
565                     createReportLink( sink, sourceDir, grammarFile, relativeOutputFileName );
566                 }
567             }
568         }
569 
570         createReportFooter( sink );
571         sink.flush();
572         sink.close();
573     }
