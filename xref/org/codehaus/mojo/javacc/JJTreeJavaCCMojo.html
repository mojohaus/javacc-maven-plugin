30  /**
31   * Preprocesses decorated grammar files (<code>*.jjt</code>) with JJTree and passes the output to JavaCC in order to
32   * finally generate a parser with parse tree actions.
33   * 
34   * @since 2.4
35   * @author Benjamin Bentmann
36   *
37   */
38  @Mojo(name = "jjtree-javacc", defaultPhase = LifecyclePhase.GENERATE_SOURCES)
39  public class JJTreeJavaCCMojo
40      extends AbstractJavaCCMojo
41  {
42  
43      /**
44       * A flag whether to generate sample implementations for <code>SimpleNode</code> and any other nodes used in the
45       * grammar. Default value is <code>true</code>.
46       *
47       */
48      @Parameter(property = "javacc.buildNodeFiles", defaultValue = "true")
49      private boolean buildNodeFiles;

51      /**
52       * A flag whether to generate a multi mode parse tree or a single mode parse tree. Default value is
53       * <code>false</code>.
54       * 
55       */

57      private boolean multi;
58  
59      /**
60       * A flag whether to make each non-decorated production void instead of an indefinite node. Default value is
61       * <code>false</code>.
62       *
63       */
64      @Parameter(property = "javacc.nodeDefaultVoid", defaultValue = "false")
65      private boolean nodeDefaultVoid;
66  
67      /**
68       * The name of a custom class that extends <code>SimpleNode</code> and will be used as the super class for the

71       *

78       * The name of a custom factory class used to create <code>Node</code> objects. This class must have a method with

82       */

86      /**

282     /**
283      * {@inheritDoc}
284      */
285     protected void processGrammar( GrammarInfo grammarInfo )
286         throws MojoExecutionException, MojoFailureException
287     {
288         File jjtFile = grammarInfo.getGrammarFile();
289         File jjtDirectory = jjtFile.getParentFile();
290 
291         File tempDirectory = getTempDirectory();
292 
293         // setup output directory of grammar file (*.jj) and node files (*.java) generated by JJTree
294         File jjDirectory = new File( tempDirectory, "node" );
295 
296         // setup output directory of parser file (*.java) generated by JavaCC
297         File parserDirectory = new File( tempDirectory, "parser" );
298 
299         // setup output directory of tree node files (*.java) generated by JJTree
300         String nodePackageName = grammarInfo.resolvePackageName( this.nodePackage );
301 
302         // generate final grammar file
303         JJTree jjtree = newJJTree();
304         jjtree.setInputFile( jjtFile );
305         jjtree.setOutputDirectory( jjDirectory );
306         jjtree.setNodePackage( nodePackageName );
307         jjtree.run();
308 
309         // generate parser files
310         JavaCC javacc = newJavaCC();
311         javacc.setInputFile( jjtree.getOutputFile() );
312         javacc.setOutputDirectory( parserDirectory );
313         javacc.run();
314 
315         // copy output from JJTree
316         copyGrammarOutput( getInterimDirectory(), ( nodePackageName != null ) ? nodePackageName
317                         : grammarInfo.getParserPackage(), jjDirectory, grammarInfo.getParserName() + "TreeConstants*" );
318 
319         // copy parser files from JavaCC
320         copyGrammarOutput( getOutputDirectory(), grammarInfo.getParserPackage(), parserDirectory,
321                            grammarInfo.getParserName() + "*" );
322 
323         // copy source files which are next to grammar unless the grammar resides in an ordinary source root
324         // (legacy support for custom sources)
325         if ( !isSourceRoot( grammarInfo.getSourceDirectory() ) )
326         {
327             copyGrammarOutput( getOutputDirectory(), grammarInfo.getParserPackage(), jjtDirectory, "*" );
328         }
329 
330         deleteTempDirectory( tempDirectory );
331     }

333     /**
334      * Creates a new facade to invoke JJTree. Most options for the invocation are derived from the current values of the
335      * corresponding mojo parameters. The caller is responsible to set the input file, output directory and package on
336      * the returned facade.
337      * 
338      * @return The facade for the tool invocation, never <code>null</code>.
339      */
340     protected JJTree newJJTree()
341     {
342         JJTree jjtree = new JJTree();
343         jjtree.setLog( getLog() );
344         jjtree.setGrammarEncoding( getGrammarEncoding() );
345         jjtree.setJdkVersion( getJdkVersion() );
346         jjtree.setStatic( getIsStatic() );
347         jjtree.setBuildNodeFiles( this.buildNodeFiles );
348         jjtree.setMulti( this.multi );
349         jjtree.setNodeDefaultVoid( this.nodeDefaultVoid );
350         jjtree.setNodeClass( this.nodeClass );
351         jjtree.setNodeFactory( this.nodeFactory );
352         jjtree.setNodePrefix( this.nodePrefix );
353         jjtree.setNodeScopeHook( this.nodeScopeHook );
354         jjtree.setNodeUsesParser( this.nodeUsesParser );
355         jjtree.setTrackTokens( this.trackTokens );
356         jjtree.setVisitor( this.visitor );
357         jjtree.setVisitorDataType( this.visitorDataType );
358         jjtree.setVisitorReturnType( this.visitorReturnType );
359         jjtree.setVisitorException( this.visitorException );
360         return jjtree;
361     }
